reserve function TextDrawing

function TextDrawing

	LoadTextFile(1, "Data/Game/Tutorial.txt")
	temp0 = object.drawX
	temp1 = object.charamount
	while temp1 > 0
	// Get the actual character at the current char pos
	GetTextInfo(object.letter, 1, TEXTINFO_TEXTDATA, Object.Index, ArrayPos0)

	Object.Frame = 0
	
	// Take uppercase letters and turn them into a matching Sprite Frame
	if object.letter > 64
		if object.letter < 91
			Object.Frame  = object.letter
			Object.Frame -= 64
		end if
	end if
	
	// Take lowercase letters and turn them into a matching Sprite Frame
	if object.letter > 96
		if object.letter < 123
			Object.Frame  = object.letter
			Object.Frame -= 96
		end if
	end if

		//ASCII Table required!
		switch object.letter
			case 33	//!
				DrawSpriteScreenFX(37, FX_INK, object.drawX, object.drawY)
				ArrayPos0++
				temp1--
				object.drawX += 0x8
				break		
			case 63	//?
				DrawSpriteScreenFX(38, FX_INK, object.drawX, object.drawY)
				ArrayPos0++
				temp1--
				object.drawX += 0x8
				break					
			case 42
				DrawSpriteScreenFX(27, FX_INK, object.drawX, object.drawY)
				ArrayPos0++
				temp1--
				object.drawX += 0x8
				break
			case 42
				// Notably, the space character (32) is skipped over here, as it shouldn't draw anything
				

				DrawSpriteScreenFX(27, FX_INK, object.drawX, object.drawY)
				

				// Move to the next letter
				ArrayPos0++
				
				// And decrease the remaining letter count, as well
				temp1--
				
				// Each letter should be spaced 8 pixels apart
				object.drawX += 0x8
				break
			case 64	//@	LINE SKIPPER
				object.drawY += 16
				if CharacterTalking == 0
					object.drawX = temp0
				else
					temp0 += 8
					object.drawX = temp0
				end if

				// Move to the next letter
				ArrayPos0++
				
				// And decrease the remaining letter count, as well
				temp1--
			break
			default
			// Notably, the space character (32) is skipped over here, as it shouldn't draw anything
			

			DrawSpriteScreenFX(Object.Frame, FX_INK, object.drawX, object.drawY)

			// Move to the next letter
			ArrayPos0++
			
			// And decrease the remaining letter count, as well
			temp1--
			
			// Each letter should be spaced 8 pixels apart
			object.drawX += 0x8
			break
		end switch
	loop
end function

event ObjectMain
	switch object.state
	case 0	//activation
		object.alpha = 0
		Object.Index = 0
		TextActive = 1
		if object[4].value0 == 0
			StopMusic()
			object.state++
		end if
	break	
	case 1	
		HUDEnabled = 0
		object.alpha += 8
		object.timer++
		object.charamount = 0
		if object.timer == 30
			object.alpha = 255
			object.timer = 0
			object.state++
		end if
	break
	case 2
		GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
		if object.charamount == temp0
			object.timer++
			if object.timer == 150
				object.timer = 0
				object.state++
			end if
		else
			object.charamount++
			PlaySfx(SfxName[Text], false)
		end if
	break
	case 3
		object.charamount = 0
		if object.alpha > -1
			object.alpha -= 8
		else
			object.alpha = 0 			
		end if	

		if crafting.state == 1
			object.alpha = 255
			object.index++
			object.state++
		end if
	break
	case 4
	if crafting.state == 1
		GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
		if object.charamount == temp0
			
		else
			object.charamount++
			PlaySfx(SfxName[Text], false)
		end if
	else
		object.alpha = 0	
		object.charamount = 0
		if crafting.state == 8
			object.index++
			BoxPositionY = 1
			CharacterTalking = 1
			object.alpha = 255
			object.state++
		end if
	end if	
	break
	case 5
	if object.index < 8
		GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
		if object.charamount == temp0
			object.timer = 0
			if inputpress[2].start == 1
				object.index++
				object.charamount = 0
			end if
		else
			object.timer++
			if object.timer == 2
				object.charamount++
				PlaySfx(SfxName[Text], false)
				object.timer = 0
			end if
		end if
	else
		TextActive = 0
		CutsceneActive = 0
		object.charamount = 0
		object.state++
	end if
	break
	case 6
		if object[6].state == 3
			TextActive = 1
			object.charamount = 0		
			object.state++
		end if
	break	
	case 7	//boss text
	if object.index < 10
		GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
		if object.charamount == temp0
			object.timer++
			if object.timer == 160
				object.index++
				object.timer = 0
				object.charamount = 0
			end if
		else
			object.timer++
			if object.timer == 2
				object.charamount++
				PlaySfx(SfxName[Text], false)
				object.timer = 0
			end if
		end if
	else
		TextActive = 0
		CutsceneActive = 0
		object.charamount = 0
		object.state++
	end if
	break
	case 8
		if object[6].state == 20

			TextActive = 1
			PlayerHealth = 3
			object[2].yvel = 0x2A000
			object[2].state = 16

			object.index = 10	
			object.charamount = 0

			object.state++
		end if
	break
	case 9

		if object[2].ypos == 0
			CutsceneActive = 1
		end if	

		if object.index < 15
			GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
			if object.charamount == temp0
				object.timer = 0
				if inputpress[2].start == 1
					object.index++
					object.charamount = 0
				end if
			else
				object.timer++
				if object.timer == 2
					object.charamount++
					PlaySfx(SfxName[Text], false)
					object.timer = 0
				end if
			end if
		else
			TextActive = 0
			CutsceneActive = 0
			object.charamount = 0
			object.state++
		end if
	break	
	case 10
		if object[6].state == 23
			TextActive = 1
			object.timer = 0
			object.index = 15
			object.state++
		end if
	break
	case 11
		if object.index < 17
			GetTextInfo(temp0, 1, TEXTINFO_TEXTSIZE, Object.Index, 0)
			if object.charamount == temp0
				object.timer++
				if object.timer == 150
					object.timer = 0
					object.index++
					object.charamount = 0
				end if
			else
				object.timer++
				if object.timer == 2
					object.charamount++
					PlaySfx(SfxName[Text], false)
					object.timer = 0
				end if
			end if
		else
			TextActive = 0
			CutsceneActive = 0
			object.charamount = 0
			object.state++
		end if
	break
	case 12
		object.timer++
		if object.timer == 200
			object.timer = 0
			object[2].controlmode = -1
			PlayMusic(6)
			object.state++
		end if
	break
	case 13
		object.timer++
		if object.timer == 300

			stage.activeList = PRESENTATION_STAGE
			stage.listPos = 6
			LoadStage()						
		
		end if
	break
	end switch
end event

event ObjectDraw
	object.inkEffect = INK_ALPHA

	ArrayPos0 = 0	//starting letter



	if TextActive == 1

		if BoxPositionY == 0
		
			object.drawY = 35
			
			temp1 = screen.xcenter

			DrawSpriteScreenFX(36, FX_INK, temp1, object.drawY)

		else

			object.drawY = 180

			object.boxYpos = object.drawY
			object.boxYpos -= 25

			temp1 = screen.xcenter
			DrawSpriteScreenFX(36, FX_INK, temp1, object.drawY)

			if CharacterTalking == 1
				temp1 += 36
				DrawSpriteScreenFX(35, FX_INK, temp1, object.drawY)
			end if	

		end if

		if CharacterTalking == 1
			temp3 = screen.xcenter
			temp3 -= 64
			object.drawX = temp3
		else
			temp3 = screen.xcenter
			temp3 -= 138
			object.drawX = temp3
		end if

		CallFunction(TextDrawing)
	end if	

	if object.state >= 13
		DrawRect(0, 0, screen.xsize, screen.ysize, color7, color7, color7, 255)

		temp1 = screen.xcenter

		DrawSpriteScreenFX(39, FX_INK, temp1, screen.ycenter)

	end if



end event
