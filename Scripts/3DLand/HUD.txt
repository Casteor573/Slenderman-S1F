function PlayerHPStandalone

		temp1 = 65
		temp2 = screen.ycenter
		temp2 -= 0x50

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 3
			DrawSpriteScreenXY(34, temp1, temp2)
		end if

		temp1 += 0x18

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 2
			DrawSpriteScreenXY(34, temp1, temp2)
		end if

		temp1 += 0x18

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 1
			DrawSpriteScreenXY(34, temp1, temp2)
		end if

end function

function CraftingUI
		DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 180)

		//THE BOOKS
		switch Crafting.State
		case 0
			//if it isnt already
			PlayerHealth = 3
			crafting.state++
		break
		case 1
			object.timer++
			if object.timer == 270
				CheckEqual(screen.xsize, 426)
				temp0 = checkResult
				if temp0 == 1
					PlayMusic(2)
					object.timer = 0
					crafting.state++
				else
					HasChainSaw = 1
					StopMusic()
					object.timer = 0
					PlaySfx(SfxName[sans], 0)
					crafting.state = 7
				end if
			end if
		break
		case 2	//first open
			temp3 = screen.xcenter
			DrawSpriteScreenXY(36, temp3, screen.ycenter)//the board


			temp0=screen.xcenter
			temp0-=88
			CheckTouchRect(temp0, 64, 160, 96)
			if checkResult > -1
				temp1 = screen.xcenter
				temp1 -= 77
				temp2 = screen.ycenter
				temp2 -= 48
				DrawSpriteScreenXY(37, temp1, temp2)
				Crafting.State++
			PlaySfx(SfxName[Score Add], 0)				
			end if

		break
		case 3	//opened book
			temp3 = screen.xcenter	
			temp3 += 72
			DrawSpriteScreenXY(36, temp3, screen.ycenter)//the board
			DrawSpriteScreenXY(39, temp3, screen.ycenter)//the reciepe book open


			CheckTouchRect(52, 72, 82, 96)
			if checkResult > -1
				PlaySfx(SfxName[Score Add], 0)	
				Crafting.State++			
			end if
		break
		case 4
			temp3 = screen.xcenter
			temp3 += 72
			DrawSpriteScreenXY(36, temp3, screen.ycenter)//the board
			DrawSpriteScreenXY(39, temp3, screen.ycenter)//the reciepe book open

			DrawSpriteScreenXY(42, temp3, screen.ycenter)//recipe


			CheckTouchRect(320, 64, 350, 96)
			if checkResult > -1
				PlaySfx(SfxName[Lamp Post], 0)
				Crafting.State++			
			end if
			
		break
		case 5
			temp3 = screen.xcenter
			temp3 += 72
			DrawSpriteScreenXY(36, temp3, screen.ycenter)//the board
			DrawSpriteScreenXY(39, temp3, screen.ycenter)//the reciepe book open

			DrawSpriteScreenXY(44, temp3, screen.ycenter)//recipe

			DrawSpriteScreenXY(43, touchscreen[0].xpos, touchscreen[0].ypos)
		
			CheckTouchRect(202, 170, 234, 202)
			if checkResult > -1
				PlaySfx(SfxName[Score Add], 0)	
				HasChainSaw = 1	
				crafting.state++
			end if		



		
		
		break
		case 6
			temp3 = screen.xcenter
			temp3 += 72
			DrawSpriteScreenXY(36, temp3, screen.ycenter)//the board
			DrawSpriteScreenXY(39, temp3, screen.ycenter)//the reciepe book open

			DrawSpriteScreenXY(44, temp3, screen.ycenter)//recipe

			temp3 -= 74
			temp4 = screen.ycenter
			temp4 += 60
			DrawSpriteScreenXY(45, temp3, temp4)//recipe



			object.timer++
			if object.timer == 100
				object.timer = 0
				StopMusic()
				PlaySfx(SfxName[sans], 0)
				HasChainSaw = 1
				crafting.state++
			end if

		break
		case 7
			DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 255)
			object.timer++
			if object.timer == 50
				object.timer = 0
				HasChainSaw = 1 //amkng sure
				PlaySfx(SfxName[sans], 0)
				MouseEnabled = 0 
				PlayMusic(3)
				crafting.state++
			end if			
		break
		case 8

		break
		end switch
	
		//NUMBERS
		
		if crafting.state > 1
			if crafting.state < 7
				temp1 = screen.xcenter
				if Crafting.State >= 3
					temp1 += 16
				else
					temp1 -= 56
				end if
				temp2 = screen.ycenter
				temp2 += 69
				if Crafting.State < 4	//12	
					DrawSpriteScreenXY(40, temp1, temp2)
				else
					DrawSpriteScreenXY(41, temp1, temp2)
				end if
			end if	
		end if	


end function

function MainHUD
		//FLASHLIGHT
		UseFlashlight = 1
		if ThunderActive == 0
			if UseFlashlight == 1
				if FlashBattery > 0	//battery greater then 0?
					if object[2].controlmode != -1	//can the player move?
						if inputdown[2].buttonY == 0	//if your holding the button
							DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 205)
						else
						
							if inputpress[2].buttonY == 1
								PlaySfx(SfxName[Flashlight], 0)
							end if

							FlashBattery -= 1

							temp0=128
							temp0*=Object.Scale
							temp0>>=9
							

							temp5=256
							temp5*=Object.Scale
							temp5>>=9

							temp1=screen.xcenter
							temp1-=temp0

							temp2=temp1
							temp2+=temp5

							temp3=120
							temp3-=temp0

							temp4=temp3
							temp4+=temp5

							DrawRect(0, 0, temp1, screen.ysize, 0x00, 0x00, 0x00, 205)
							DrawRect(temp2, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 205)
							DrawRect(temp1, 0, temp5, temp3, 0x00, 0x00, 0x00, 205)
							DrawRect(temp1, temp4, temp5, screen.ysize, 0x00, 0x00, 0x00, 205)
							
							temp6 = screen.xsize
							temp6 -= screen.xcenter
							
							DrawSpriteScreenFX(15,FX_INK,temp6,screen.ycenter)
							object.InkEffect = INK_ALPHA
							object.alpha = 200
						end if		
					else
						DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 205)	
					end if	
				else	//out of power ur cooked
					DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 224)

					if inputpress[2].buttonY == 1
						PlaySfx(SfxName[Error], 0)
					end if
				end if		
			end if		
		else
			DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 160)	
		end if	


		//BATTERY
	
		temp3 = 210
	
		DrawSpriteScreenXY(17, temp3, screen.ycenter)

		//
		if FlashBattery > 4000
			DrawSpriteScreenXY(18, temp3, screen.ycenter)
		end if

		if FlashBattery > 3000
			DrawSpriteScreenXY(19, temp3, screen.ycenter)
		end if	

		if FlashBattery > 2000
			DrawSpriteScreenXY(20, temp3, screen.ycenter)
		end if		

		if FlashBattery > 1000
			DrawSpriteScreenXY(21, temp3, screen.ycenter)
		end if	

		if FlashBattery > 0
			DrawSpriteScreenXY(22, temp3, screen.ycenter)
		end if	

		//HEALTH

		temp1 = 65
		temp2 = screen.ycenter
		temp2 -= 0x50

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 3
			DrawSpriteScreenXY(34, temp1, temp2)
		end if

		temp1 += 0x18

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 2
			DrawSpriteScreenXY(34, temp1, temp2)
		end if

		temp1 += 0x18

		DrawSpriteScreenXY(35, temp1, temp2)

		if PlayerHealth >= 1
			DrawSpriteScreenXY(34, temp1, temp2)
		end if



		// Find the initial X Position to draw the HUD elements
		temp1 = 95
		
		// Draw the "UFO" graphic
		DrawSpriteScreenXY(10, temp1, 20)
		
		// Draw the first digit of the UFOs counter
		temp0 = Object.PageCount
		temp0 /= 10
		temp1 += 24
		DrawSpriteScreenXY(temp0, temp1, 21)
		
		// Draw the second digit of the UFOs counter
		temp0 = Object.PageCount
		temp0 %= 10
		temp1 += 8
		DrawSpriteScreenXY(temp0, temp1, 21)

end function

function TouchControls
	object.inkEffect = INK_ALPHA

	temp0 = screen.xsize
	temp0 -= 44
	DrawSpriteScreenXY(49,  26,1 49) //dpad
	temp0 -= 39

	DrawSpriteScreenXY(56, 0, temp0, 8) //pause
	
	
	//movement
	
	temp0 -= 10
	if inputDown.Left==1
		DrawSpriteScreenXY(52,  26, 151)
	endif
	if inputDown.Down==1
		DrawSpriteScreenXY(51,  26, 189)
	end if
	if inputDown.Right==1
		DrawSpriteScreenXY(53,  64, 151)
	endif
	if inputDown.Up==1
		DrawSpriteScreenXY(50,  26, 151)
	endif
	
	temp0 += 15	
	//jump
	if inputDown.buttonA==1
		DrawSpriteScreenXY(55,  temp0, 163)	
	else
		DrawSpriteScreenXY(54,  temp0, 163)
	endif

	if UseFlashlight == 1
		temp0 -= 50
		//flashlight
		if inputDown.buttonY==1
			DrawSpriteScreenXY(58,  temp0, 54)	
		else
			DrawSpriteScreenXY(57,  temp0, 54)
		endif	
	end if

	temp0 += 80

	if CharacterTalking == 1
		DrawSpriteScreenXY(60,  temp0, 12)
	else
		DrawSpriteScreenXY(59,  temp0, 12)
	end if
end function



event ObjectDraw
//Not Dead
if HUDEnabled == 1
	if object[2].state != 13
		CallFunction(MainHUD)
	else	
		//Dead
		DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 224)
	end if
else
	if CutsceneActive == 1
		CallFunction(CraftingUI)
	else
		DrawRect(0, 0, screen.xsize, screen.ysize, 0x00, 0x00, 0x00, 180)
		//not dead	
		if object[2].state != 13
			CallFunction(PlayerHPStandalone)
		end if
	end if
end if

	//jumpscare


	//touch controls
	if saved.mobile == 1
		CallFunction(TouchControls)
		DrawSpriteScreenXY(16, touchscreen[0].xpos, touchscreen[0].ypos)
	end if


	if MouseEnabled == 1
		DrawSpriteScreenXY(16, touchscreen[0].xpos, touchscreen[0].ypos)
	end if

	if object[6].state > 3
		if object[2].state == 13
			object.deathtimer++
			if object.deathtimer == 1
				PlayMusic(5)
			end if	
		end if
	end if


	//debugging


	if stage.debugmode == 1

		//slenderman state
		DrawNumbers(0, 8, 25, object[6].state, 3, 8, 0)
		//slenderman timer
		DrawNumbers(0, 16, 48, object[6].value3, 3, 8, 0)

		DrawNumbers(0, 16, 72, SlenderHealth, 3, 8, 0)

		DrawNumbers(0, 16, 96, object[6].value7, 3, 8, 0)

	end if

end event
